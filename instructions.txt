PascalCase -> messages
snake_case -> field/attributes
UPPER_CASE -> enum values
camelCase -> func names

python -m grpc_tools.protoc ^ -I . ^ --python_out=. ^ --grpc_python_out=. ^ proto\common.proto ^ proto\namenode.proto ^ proto\datanode.proto

for namenode: python -m namemnode.app.main
